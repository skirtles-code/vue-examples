Toggle Switch #

Toggle Switch Example #

Live Example
Bound value: true
<template>
  <div
    class="toggle-switch"
    :class="{ checked: modelValue }"
    @click="toggle"
  />
</template>

<script setup>
const props = defineProps({
  modelValue: {
    required: true
  }
})

const emit = defineEmits(['update:modelValue'])

const toggle = () => {
  emit('update:modelValue', !props.modelValue)
}
</script>
